query AllUser {
  users {
    username
    points
    habits {
      name
      days {
        date
      }
      _id
    }
  }
}

query loginUser($input: loginInput) {
  login(input: $input) {
    username
    points
    token
    
  }
}

mutation registerUser($input: loginInput) {
  register(input: $input) {
    username
    points
    token
  }
}

mutation addHabit($name: String!, $color: String!) {
  addHabit(name: $name, color: $color) {
    name
    color
    owner {
      username
    }
  }
}

mutation deleteHabit($habitId: String!) {
  deleteHabit(habitId: $habitId)
}

mutation addNDays($habitId: String!, $num: Int!) {
  addNDays(habitId: $habitId, num: $num)
}
